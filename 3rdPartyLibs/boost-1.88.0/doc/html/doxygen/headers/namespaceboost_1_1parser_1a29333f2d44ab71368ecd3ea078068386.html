Function template delimiter

boost::parser::delimiter

Synopsis

// In header: <boost/parser/parser.hpp>


template<typename DelimiterParser> 
  delimiter_directive< DelimiterParser > 
  delimiter(parser_interface< DelimiterParser > delimiter_parser);

Description

Returns a delimiter_directive whose operator[] returns a perm_parser, where the items parsed are delimited by delimiter_parser.
