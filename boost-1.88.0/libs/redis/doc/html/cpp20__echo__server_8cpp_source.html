25auto
26echo_server_session(
27 asio::ip::tcp::socket socket,
28 std::shared_ptr<connection> conn) -> asio::awaitable<void>
29{
30 request req;
31 response<std::string> resp;
32
33 for (std::string buffer;;) {
34 auto n = co_await asio::async_read_until(socket, asio::dynamic_buffer(buffer, 1024), "\n");
35 req.push("PING", buffer);
36 co_await conn->async_exec(req, resp);
37 co_await asio::async_write(socket, asio::buffer(std::get<0>(resp).value()));
38 std::get<0>(resp).value().clear();
39 req.clear();
40 buffer.erase(0, n);
41 }
42}

44// Listens for tcp connections.
45auto listener(std::shared_ptr<connection> conn) -> asio::awaitable<void>
46{
47 try {
48 auto ex = co_await asio::this_coro::executor;
49 asio::ip::tcp::acceptor acc(ex, {asio::ip::tcp::v4(), 55555});
50 for (;;)
51 asio::co_spawn(ex, echo_server_session(co_await acc.async_accept(), conn), asio::detached);
52 } catch (std::exception const& e) {
53 std::clog << "Listener: " << e.what() << std::endl;
54 }
55}
56
57// Called from the main function (see main.cpp)
58auto co_main(config cfg) -> asio::awaitable<void>
59{
60 auto ex = co_await asio::this_coro::executor;
61 auto conn = std::make_shared<connection>(ex);
62 asio::co_spawn(ex, listener(conn), asio::detached);
63 conn->async_run(cfg, {}, asio::consign(asio::detached, conn));
64
65 signal_set sig_set(ex, SIGINT, SIGTERM);
66 co_await sig_set.async_wait();
67 conn->cancel();
68}
